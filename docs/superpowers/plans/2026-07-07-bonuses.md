```java
package com.akamai.wsa.domain.model;

import java.time.Duration;
import java.time.Instant;

public enum Interval {
    ONE_MINUTE("1m", Duration.ofMinutes(1)),
    FIVE_MINUTES("5m", Duration.ofMinutes(5)),
    ONE_HOUR("1h", Duration.ofHours(1));

    private final String label;
    private final Duration duration;

    Interval(String label, Duration duration) {
        this.label = label;
        this.duration = duration;
    }

    public String label() {
        return label;
    }

    public Duration duration() {
        return duration;
    }

    public static Interval fromLabel(String label) {
        for (Interval interval : values()) {
            if (interval.label.equals(label)) {
                return interval;
            }
        }
        throw new IllegalArgumentException("unsupported interval: " + label);
    }

    public Instant floor(Instant instant) {
        long stepSeconds = duration.getSeconds();
        long flooredEpochSecond = Math.floorDiv(instant.getEpochSecond(), stepSeconds) * stepSeconds;
        return Instant.ofEpochSecond(flooredEpochSecond);
    }
}
```

```java
package com.akamai.wsa.application.timeseries;

import com.akamai.wsa.domain.model.Interval;
import com.akamai.wsa.domain.model.SecurityEvent;
import com.akamai.wsa.domain.port.EventStore;
import org.springframework.stereotype.Service;

import java.time.Instant;
import java.util.List;
import java.util.Map;
import java.util.TreeMap;

@Service
public class TimeSeriesService implements BuildTimeSeries {

    private final EventStore eventStore;

    public TimeSeriesService(EventStore eventStore) {
        this.eventStore = eventStore;
    }

    @Override
    public TimeSeries build(TimeSeriesQuery timeSeriesQuery) {
        Interval interval = timeSeriesQuery.interval();
        Map<Instant, Long> counts = new TreeMap<>();
        for (SecurityEvent securityEvent : eventStore.findMatching(timeSeriesQuery.configId(), timeSeriesQuery.timeRange())) {
            Instant bucketStart = interval.floor(securityEvent.timestamp());
            counts.merge(bucketStart, 1L, Long::sum);
        }
        List<TimeBucket> buckets = counts.entrySet().stream()
                .map(entry -> new TimeBucket(entry.getKey(), entry.getValue()))
                .toList();
        return new TimeSeries(timeSeriesQuery.configId(), timeSeriesQuery.timeRange(), interval, buckets);
    }
}
```

```java
package com.akamai.wsa.infrastructure.ratelimit;

import java.time.Clock;
import java.time.Instant;
import java.util.concurrent.ConcurrentHashMap;

public class RateLimiter {

    private record Window(Instant windowStart, int count) {
    }

    private final RateLimitProperties properties;
    private final Clock clock;
    private final ConcurrentHashMap<String, Window> windowsByClient = new ConcurrentHashMap<>();

    public RateLimiter(RateLimitProperties properties, Clock clock) {
        this.properties = properties;
        this.clock = clock;
    }

    public boolean tryAcquire(String clientIdentifier) {
        Instant now = clock.instant();
        Window updated = windowsByClient.compute(clientIdentifier, (key, existing) -> {
            if (existing == null || now.isAfter(existing.windowStart().plus(properties.window()))) {
                return new Window(now, 1);
            }
            return new Window(existing.windowStart(), existing.count() + 1);
        });
        return updated.count() <= properties.maxRequests();
    }
}
```
